import React, { useState } from 'react';
import { useAuth } from '../context/AuthContext';
import { useSettings } from '../context/SettingsContext';
import { ShieldAlert, KeyRound } from 'lucide-react';

export function Settings() {
  const { user } = useAuth();
  const { enableDeletion, toggleDeletion } = useSettings();
  const [password, setPassword] = useState('');
  const [error, setError] = useState('');

  if (user?.role !== 'super_admin') {
    return <div className="p-8 text-center text-slate-500">Access Denied. Super Admin only.</div>;
  }

  const handleToggle = (e: React.FormEvent) => {
    e.preventDefault();
    setError('');
    
    if (password === 'admin123') {
      toggleDeletion();
      setPassword('');
    } else {
      setError('Incorrect password. Please try again.');
    }
  };

  return (
    <div className="max-w-2xl mx-auto space-y-6">
      <div className="card p-6">
        <h2 className="text-xl font-semibold text-slate-800 flex items-center mb-6 border-b border-slate-100 pb-4">
          <ShieldAlert className="w-5 h-5 mr-3 text-slate-400" />
          System Settings
        </h2>

        <div className="space-y-6">
          <div className="bg-slate-50 p-4 rounded-lg border border-slate-200">
            <div className="flex items-center justify-between mb-4">
              <div>
                <h3 className="font-medium text-slate-800">Global Delete Functionality</h3>
                <p className="text-sm text-slate-500 mt-1">
                  When enabled, Super Admin users can delete records (Sales, Purchases, Customers, Vendors, Branches, Payroll, Ledger).
                </p>
                <div className="mt-2 text-xs font-semibold px-2 py-1 bg-slate-200 text-slate-700 rounded inline-block">
                  Current Status: <span className={enableDeletion ? 'text-red-600' : 'text-emerald-600'}>{enableDeletion ? 'ENABLED' : 'DISABLED'}</span>
                </div>
              </div>
            </div>

            <form onSubmit={handleToggle} className="mt-4 pt-4 border-t border-slate-200">
              <label className="block text-sm font-medium text-slate-700 mb-2">
                Enter password to {enableDeletion ? 'disable' : 'enable'} overrides
              </label>
              <div className="flex space-x-3">
                <div className="relative flex-1">
                  <div className="absolute inset-y-0 left-0 pl-3 flex items-center pointer-events-none">
                    <KeyRound className="h-4 w-4 text-slate-400" />
                  </div>
                  <input
                    type="password"
                    value={password}
                    onChange={(e) => setPassword(e.target.value)}
                    className="pl-10 input-field"
                    placeholder="Security Password"
                    required
                  />
                </div>
                <button
                  type="submit"
                  className={`px-4 py-2 rounded text-sm font-medium transition-colors ${
                    enableDeletion 
                    ? 'bg-slate-800 text-white hover:bg-slate-900 shadow-md' 
                    : 'bg-red-500 text-white hover:bg-red-600 shadow-md border border-red-600'
                  }`}
                >
                  {enableDeletion ? 'Disable' : 'Enable'}
                </button>
              </div>
              {error && <p className="text-rose-500 text-sm mt-2">{error}</p>}
            </form>
          </div>
        </div>
      </div>
    </div>
  );
}
